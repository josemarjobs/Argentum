175 	private void carregarDados() {
176 		List<Negocio> negocios = new EscolheXML().escolhe();
177 		filtrarPorData(negocios);
178 		NegociosTableModel ntm = new NegociosTableModel(negocios);
179 		ReflectionTableModel rtm=new ReflectionTableModel(negocios);
180 		tabela.setModel(rtm);
181 
182 		CandlestickFactory candlestickFactory = new CandlestickFactory();
183 
184 		List<Candle> candles = candlestickFactory.construirCandles(negocios);
185 
186 		SerieTemporal serie = new SerieTemporal(candles);
187 
188 		GeradorDeGraficos geradorDeGraficos = new GeradorDeGraficos(serie, 2,
189 				serie.getTotal() - 1);
190 		geradorDeGraficos.criaGrafico("Media Movel Simples");
191 
192 		if (mediaFechamento.isSelected()) {
193 			geradorDeGraficos.plotarGrafico(new MediaMovelSimples(
194 					new IndicadorDeFechamento()));
195 		}
196 		
197 		if(mediaAbertura.isSelected()){
198 			geradorDeGraficos.plotarGrafico(new MediaMovelSimples(
199 					new IndicadorDeAbertura()));
200 		}
201 		
202 		JPanel grafico = geradorDeGraficos.getPanel();
203 		this.abas.setComponentAt(1, grafico);
204 	}

206 	private void filtrarPorData(List<Negocio> negocios) {
207 		try {
208 			String valor = (String) campoDataInicio.getValue();
209 			SimpleDateFormat formato = new SimpleDateFormat("dd/MM/yyyy");
210 			formato.setLenient(false);
211 			Date dataInicio = formato.parse(valor);
212 
213 			Iterator<Negocio> it = negocios.iterator();
214 			while (it.hasNext()) {
215 				if (it.next().getData().getTime().before(dataInicio)) {
216 					it.remove();
217 				}
218 			}
219 
220 		} catch (Exception e) {
221 			campoDataInicio.setValue(null);
222 		}
223 	}
